164 Eigen::Matrix< T, 1, Eigen::Dynamic > LogScaleVector( T lower_bound, T upper_bound, uint num_elements ) {
165 
166  T min_elem = static_cast<T>( log10(lower_bound) );
167  T max_elem = static_cast<T>( log10(upper_bound) );
168  T delta = max_elem - min_elem;
169 
170  Eigen::Matrix< T, 1, Eigen::Dynamic > log_space_vector;
171  log_space_vector.resize( num_elements );
172 
173  for ( uint i = 0; i < num_elements ; i ++ ) {
174 
175  T step = static_cast<T>( i )/static_cast<T>( num_elements - 1 );
176  auto lin_step = delta*step + min_elem;
177 
178  log_space_vector( 0, i ) = static_cast<T>( std::pow( 10.0, lin_step ) );
179  }
180 
181  return log_space_vector;
182 }

195 bool FOS<T>::ComputeStatsCond( uint stats_it, T r_stats_it, const Eigen::Matrix< T, 1, Eigen::Dynamic >& lambdas ) {
196 
197  bool stats_cond = true;
198 
199  for ( uint i = 1; i <= stats_it; i++ ) {
200 
201  Eigen::Matrix< T, Eigen::Dynamic, 1 > beta_k = Betas.col( i - 1 );
202  T rk = lambdas( 0, i - 1 );
203 
204  Eigen::Matrix< T, Eigen::Dynamic, 1 > beta_diff = Betas.col( stats_it - 1 );
205  beta_diff -= beta_k;
206  T abs_max_betas = beta_diff.cwiseAbs().maxCoeff();
207 
208  T n = static_cast<T>( X.rows() );
209 
210  T check_parameter = n*abs_max_betas / ( r_stats_it + rk );
211 
212  DEBUG_PRINT( "Check Parameter: " << check_parameter );
213 
214  stats_cond &= ( check_parameter <= C );
215  }
216 
217  return stats_cond;
218

268 template < typename T >
269 T dual_objective ( const Eigen::Matrix< T, Eigen::Dynamic, Eigen::Dynamic >& X, \
270  const Eigen::Matrix< T, Eigen::Dynamic, 1 >& Y, \
271  const Eigen::Matrix< T, Eigen::Dynamic, 1 >& Beta, \
272  T r_stats_it ) {
273 
274  Eigen::Matrix< T, Eigen::Dynamic, 1 > error = X*Beta - Y;
275 
276  Eigen::Matrix< T, Eigen::Dynamic, 1 > alt_part_0 = 2.0f*X.transpose()*error;
277  //Compute dual point
278  T alternative = r_stats_it/( alt_part_0.template lpNorm < Eigen::Infinity >() );
279 
280  T alt_part_1 = -1.0*static_cast<T>( Y.transpose()*error );
281  Eigen::Matrix< T, Eigen::Dynamic, 1 > alt_part_2 = Y - X*Beta;
282 
283  T alternative_0 = alt_part_1/( alt_part_2.squaredNorm() );
284 
285  T s = std::min( std::max( alternative, alternative_0 ), -1.0f*alternative );
286 
287  Eigen::Matrix< T, Eigen::Dynamic, 1 > nu_t = -1.0f*( 2.0f*s / r_stats_it ) * error;
288 
289  Eigen::Matrix< T, Eigen::Dynamic, 1 > nu_part = nu_t + 2.0f/r_stats_it*Y;
290 
291  T d_nu = 0.25* square( r_stats_it )*nu_part.squaredNorm() - Y.squaredNorm();
292 
293  return d_nu;
294 }
