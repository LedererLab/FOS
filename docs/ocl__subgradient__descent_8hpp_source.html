153 template < typename T >
154 void SubGradientSolver< T, Base >::f_beta_tilda (
155  const cl::Buffer& X,
156  const cl::Buffer& Y,
157  const cl::Buffer& Beta,
158  const cl::Buffer& Beta_prime,
159  T L ) {
160 
161  Eigen::Matrix< T, Eigen::Dynamic, 1 > f_beta = X*Beta_prime - Y;
162  T taylor_term_0 = f_beta.squaredNorm();
163 
164  Eigen::Matrix< T, Eigen::Dynamic, 1 > f_grad = 2.0*X.transpose()*( f_beta );
165  Eigen::Matrix< T, Eigen::Dynamic, 1 > beta_diff = ( Beta - Beta_prime );
166 
167  T taylor_term_1 = f_grad.transpose()*beta_diff;
168 
169  T taylor_term_2 = L/2.0*beta_diff.squaredNorm();
170 
171  return taylor_term_0 + taylor_term_1 + taylor_term_2;
172 }
