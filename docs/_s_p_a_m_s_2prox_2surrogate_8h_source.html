128  // rho = (1-w)rho + w (rho_sample)
129  // output = (1-w)output + w(input - (1/rho) nabla(input))
130  inline void add_sample_gradient(const Vector<T>& input,
131  Vector<T>& z, T& rho, const T w) {
132  // update new_rho => (1-w)rho + w (rho_sample)
133  if (_is_normalized) {
134  // rho is already equal to genericL
135  typename U::col spw;
136  z.scal(T(1.0)-w);
137  z.add(input,w);
138  const T scal = w/(rho*_sizebatch);
139  for (int i = 0; i<_sizebatch; ++i) {
140  const int ind = _current_batch[i];
141  _Xt->refCol(ind,spw);
142  const T y= (*_y)[ind];
143  const T s = input.dot(spw);
144  z.add(spw,-scal*this->gradient_simple(y,s));
145  }
146  } else {
147  T rho_sample=0;
148  for (int i = 0; i<_sizebatch; ++i) {
149  rho_sample += _L[_current_batch[i]];
150  }
151  rho_sample /= _sizebatch;
152  rho_sample=MAX(rho_sample,T(0.1)*rho);
153  const T new_rho=(1-w)*rho + w*rho_sample;
154  //const T new_rho=rho;
155  typename U::col spw;
156  z.scal(rho*(T(1.0)-w)/new_rho);
157  z.add(input,rho_sample*w/new_rho);
158  const T scal = w/(_sizebatch*new_rho);
159  for (int i = 0; i<_sizebatch; ++i) {
160  const int ind = _current_batch[i];
161  _Xt->refCol(ind,spw);
162  const T y= (*_y)[ind];
163  const T s = input.dot(spw);
164  z.add(spw,-scal*this->gradient_simple(y,s));
165  }
166  rho=new_rho;
167  }
168  };

198  void inline add_scal_grad(Vector<T>& input, const T scalin, const T scal, Vector<T>& av, const T scal2 = 0) {
199  _tmp.resize(_sizebatch);
200  typename U::col spw;
201  if (_sizebatch==1) {
202  const int ind = _current_batch[0];
203  _Xt->refCol(ind,spw);
204  const T s = scalin*input.dot(spw);
205  const T y= (*_y)[ind];
206  const T s2 = this->gradient_simple(y,s);
207  input.add(spw,scal*s2);
208  if (scal2) av.add(spw,scal2*s2);
209  } else {
210  for (int i = 0; i<_sizebatch; ++i) {
211  const int ind = _current_batch[i];
212  _Xt->refCol(ind,spw);
213  _tmp[i] = scalin*input.dot(spw);
214  }
215  const T scalb = scal/_sizebatch;
216  const T scal2b = scal2/_sizebatch;
217  for (int i = 0; i<_sizebatch; ++i) {
218  const int ind = _current_batch[i];
219  const T y= (*_y)[ind];
220  _Xt->refCol(ind,spw);
221  const T s2 = this->gradient_simple(y,_tmp[i]);
222  input.add(spw,scalb*s2);
223  if (scal2) av.add(spw,scal2b*s2);
224  }
225  }

227  inline T scal_grad(const T s) {
228  const int ind = _current_batch[0];
229  const T y= (*_y)[ind];
230  return this->gradient_simple(y,s);

286  void inline choose_random_batch() {
287  if (_random) {
288  for (int i = 0; i<_nbatches; ++i) // size of the mini batches
289  _current_batch[i]=random() % _n;
290  } else {
291  for (int i = 0; i<_nbatches; ++i)
292  _current_batch[i]= _counter++ % _n;
293  }

296  int inline choose_random_fixedbatch() {
297  const int size_lastbatch =_nbatches+(_n-_nbatches*_num_batches);
298  _num_batch= (_random ? random() : _counter++) % _num_batches;
299  _current_batch.resize(size_lastbatch);
300  _sizebatch = _num_batch == _num_batches-1 ? size_lastbatch : _nbatches;
301  for (int i = 0; i<_sizebatch; ++i) // size of the mini batches
302  _current_batch[i]=_num_batch*_nbatches+i;
303  return _num_batch;

305  void inline getL(Vector<T>& stats) {
306  if (_constantL) {
307  for (int i = 0; i<_num_batches-1; ++i)
308  stats[i]=_genericL*_nbatches;
309  const int size_lastbatch =_nbatches+(_n-_nbatches*_num_batches);
310  stats[_num_batches-1]=_genericL*size_lastbatch;
311  } else {
312  for (int i = 0; i<_num_batches; ++i) {
313  stats[i]=0;
314  for (int j = 0; j<_nbatches; ++j) {
315  stats[i]+=_L[i*_nbatches+j];
316  }
317  }
318  const int size_lastbatch =_nbatches+(_n-_nbatches*_num_batches);
319  for (int j = _nbatches; j<size_lastbatch; ++j)
320  stats[_num_batches-1]+=_L[(_num_batches-1)*_nbatches+j];
321  }
322  };

485  virtual void update_incremental_surrogate(const Vector<T>& input) {
486  const int num_batch = _function->choose_random_fixedbatch();
487  Vector<T> z_old;
488  _stats2.refCol(num_batch,z_old);
489  const T rho_old=_stats[num_batch];
490  if (this->_strategy <= 2 || this->_strategy == 4) {
491  //if (!this->_first_pass)
492  _z.sub(z_old);
493  _function->add_sample_gradient2(input,z_old,rho_old*_scalL);
494  _z.add(z_old);
495  } else {
496  T old_value=0;
497  T old_valueb=0;
498  if (!this->_first_pass) {
499  _z.add(z_old,-rho_old);
500  _z3.copy(input);
501  _z3.sub(z_old);
502  //old_value = _stats4[num_batch] + _function->dotprod_gradient3(_z3,_stats3)+0.5*rho_old*_scalL*_z3.nrm2sq();
503  old_value = _stats4[num_batch] + _function->dotprod_gradient3(_z3,_stats3); // f_old+ nabla f(old)'( new-old)
504  old_valueb=T(0.5)*rho_old*_z3.nrm2sq();
505  }
506  z_old.copy(input);
507  _z.add(z_old,rho_old);
508  _function->add_sample_gradient3(input,_z2,_stats3,_stats4[num_batch]);
509  if (!this->_first_pass) {
510  _diff += (old_value-_stats4[num_batch]); // should be non-positive (convexity inequality)
511  _diffb+=old_valueb;
512 // _diff += (old_value-_stats4[num_batch] > 0 ? T(1.0) : -T(1.0));
513  }
514  }
515  };

517  const int p = static_cast<int>(input.n());
518  this->_strategy = strategy;
519  // _z.resize(p);
520  // _z.setZeros();
521  this->_rho = _function->n()*_function->genericL();
522  _z.copy(input);
523  _z.scal(_rho*_scalL);
524  const int num_batches = _function->num_batches();
525  const int n = _function->n();
526  this->_stats.resize(num_batches);
527  _function->getL(this->_stats);
528  this->_stats2.resize(p,num_batches,false);
529  Vector<T> col;
530  for (int i = 0; i<num_batches; ++i) {
531  this->_stats2.refCol(i,col);
532  col.copy(_z);
533  col.scal(T(1.0)/num_batches);
534  }
535  if (strategy == 3) {
536  this->_stats3.resize(n);
537  this->_stats3.setZeros();
538  this->_stats4.resize(num_batches);

540  _z2.resize(p);
541  _z2.setZeros();
542  }

546  output.copy(_z);
547  output.scal(T(1.0)/(_scalL*_rho));
548  } else {
549  output.copy(_z);
550  output.add(_z2,T(1.0)/_scalL);
551  output.scal(T(1.0)/(_rho));
552  }
553  };

588 
589  public:
590  ProximalSurrogate(SmoothFunction<T,U>* function,
591  Regularizer<T>* prox, const T lambda) :

602  virtual void minimize_incremental_surrogate(Vector<T>& output) {
603  const int n = this->_function->n();
604  _prox->linearize(output);

962  const bool auto_params, const int averaging_mode, const bool verbose, const bool evaluate) {
963  Timer time;
964  time.start();
965  w.copy(w0);
966  this->_logs[2]=0;
967  if (verbose && iters > 0)
968  cout << "Dedicated L1-Sparse Solver" << endl;
969 
970  if (auto_params && iters > 0) this->auto_parameters(w0,w,wav,averaging_mode);
971  const int n = _function->n();
972  const int p = _function->p();
973  const T rho=_function->genericL();
974  Vector<T> sumw(n);
975  sumw[0]=0;
976  Vector<T> prod(n);
977  prod[0]=T(1.0);
978 
979  SpVector<T> col;
980  Triplet<T,T,int>* pr_t = new Triplet<T,T,int>[p];
981  memset(pr_t,0,p*sizeof(Triplet<T,T,int>));
982 
983  int counter=0;
984  const T lambda_d_rho=_lambda/rho;
985  int renorm2=0;
986  int forgetting_offset=-1;
987  int middle_offset=0;
988  for (int t = 1; t<= iters; ++t) {
989  const T weight=this->t_to_weight(t);
990  _function->choose_random_batch();
991  const T onemw=T(1.0)-weight;
992  const int next_counter=(counter + 1) % n;
993  sumw[next_counter]= sumw[counter]+weight*lambda_d_rho;
994  if (sumw[next_counter] > T(1e50)) {
995  sumw.add(-sumw[next_counter]);
996  }
997  //const int prev_counter = (n + counter - 1) % n;
998  prod[next_counter] = (t==1) ? T(1.0) : prod[counter]*onemw;
999  if (prod[next_counter] < T(1e-8) || forgetting_offset == next_counter) {
1000  T scal=T(1.0)/prod[next_counter];
1001  for (int i = middle_offset; i != next_counter; i = (i+1) %n)
1002  prod[i] *= scal;
1003  prod[next_counter]=T(1.0);
1004  forgetting_offset=middle_offset;
1005  middle_offset=next_counter;
1006  renorm2++;
1007  }
1008 
1009  const T thrs1 = onemw*lambda_d_rho;
1010  const T thrs2 = weight/rho;
1011 
1012  _function->refData(col); // might be counter instead
1013  T* v = col.rawX();
1014  INTM * r = col.rawR();
1015  const int L = static_cast<int>(col.L());
1016  T s = 0;
1017 
1018  for (int i = 0; i< L; ++i) {
1019  const int ind=static_cast<int>(r[i]);
1020  T& prx = pr_t[ind].x;
1021  T& prz = pr_t[ind].z;
1022  const int& prs = pr_t[ind].s;
1023  const T val =abs<T>(prx);
1024  const bool negval = prx < 0;
1025  const int sc = prs;
1026  if (val) {
1027  if (sc==counter) {
1028  prz = negval ? prx - thrs1 : prx + thrs1;
1029  s+=prx*v[i];
1030  } else {
1031  const T val2 = val + sumw[sc];
1032  const T new_val = val2 - sumw[counter];
1033  T& xval=prx;
1034  if (new_val > 0) {
1035  xval = negval ? -new_val : new_val;
1036  prz =new_val + thrs1;
1037  s+=xval*v[i];
1038  } else {
1039  xval=0;
1040  // test if result is zero.
1041  int up = counter + n;
1042  int down = sc <= counter ? sc + n : sc;
1043  bool proceed = true;
1044  const int forget_offset = forgetting_offset <= counter ? forgetting_offset +n : forgetting_offset;
1045  if (sc <= forget_offset) {
1046  if (val2 <= sumw[forgetting_offset]) {
1047  prz=0;
1048  proceed = false;
1049  } else {
1050  down = forget_offset;
1051  }
1052  }
1053  if (proceed) {
1054  while ( up - down > 1) {
1055  int current = (up+down)/2;
1056  assert(current != up);
1057  assert(current != down);
1058  if (val2 <= sumw[current % n]) {
1059  up=current; // always satisfied for up
1060  } else {
1061  down=current; // never satisfied for down
1062  }
1063  }
1064  const T z_jm1= val2 - sumw[(n + up-1) % n] + lambda_d_rho;
1065  prz = z_jm1 * (prod[next_counter]/prod[up % n]);
1066  }
1067  }
1068  if (negval) prz=-prz;
1069  }
1070  } else {
1071  const int down = sc <= counter ? sc + n : sc;
1072  const int forget_offset = forgetting_offset <= counter ? forgetting_offset +n : forgetting_offset;
1073  prz = down <= forget_offset ? 0 : prz*(prod[next_counter]/prod[sc]);
1074  }
1075  }
1076  s = _function->scal_grad(s);
1077  s *= thrs2;
1078 
1082  counter = next_counter;
1083  for (int i = 0; i< L; ++i) {
1084  const int ind=static_cast<int>(r[i]);
1085  T& prx = pr_t[ind].x;
1086  T& prz = pr_t[ind].z;
1087  int& prs = pr_t[ind].s;
1088  prz -= s*v[i];
1089  if (prz > lambda_d_rho) {
1090  prx=prz-lambda_d_rho;
1091  } else if (prz < -lambda_d_rho) {
1092  prx=prz+lambda_d_rho;
1093  } else {
1094  prx=0;
1095  }
1096  prs = counter;
1097  }
1098 
1099  if (t==iters) {
1100  for (int i = 0; i<p; ++i) {
1101  T& prx = pr_t[i].x;
1102  const int& prs = pr_t[i].s;
1103  if (prx && counter != prs) {
1104  const T diff=sumw[counter] - sumw[prs];
1105  if (prx > diff) {
1106  prx -= diff;
1107  } else if (prx < -diff) {
1108  prx += diff;
1109  } else {
1110  prx =0;
1111  }
1112  }
1113  }
1114  }
1115  }
1116 
1117  for (int i = 0; i<p; ++i) {
1118  w[i]=pr_t[i].x;
1119  }
1120  delete[](pr_t);
1121  time.stop();
1122  if (evaluate) {
1123  this->_logs[0]=_function->eval(w)+_lambda*w.asum();
1124  this->_logs[1]=0;
1125  }
1126  this->_logs[2]=time.getElapsed();
