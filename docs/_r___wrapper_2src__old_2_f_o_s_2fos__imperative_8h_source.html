104 T dual_objective ( const Eigen::Matrix< T, Eigen::Dynamic, Eigen::Dynamic >& X, \
105  const Eigen::Matrix< T, Eigen::Dynamic, 1 >& Y, \
106  const Eigen::Matrix< T, Eigen::Dynamic, 1 >& Beta, \
107  T r_stats_it ) {
108

110 
111  //Compute dual point
112 
113  T alternative = r_stats_it/compute_lp_norm( 2.0f*X.transpose()*error, Eigen::Infinity );
114 
115  T alt_part_1 = -1.0*static_cast<T>( Y.transpose()*error );
116  Eigen::Matrix< T, Eigen::Dynamic, 1 > alt_part_2 = Y - X*Beta;
117 
118  T alternative_0 = alt_part_1/( alt_part_2.squaredNorm() );
119 
120  T s = std::min( std::max( alternative, alternative_0 ), -1.0f*alternative );
121
