167 bool X_FOS<T>::ComputeStatsCond( T C,
168  uint stats_it,
169  T r_stats_it,
170  const std::vector <T>& lambdas,
171  const MatrixT& Betas ) {
172 
173  bool stats_cond = true;
174 
175  for ( uint i = 1; i <= stats_it; i++ ) {
176 
177  VectorT beta_k = Betas.col( i - 1 );
178  T rk = lambdas.at( i - 1 );
179 
180  VectorT beta_diff = Betas.col( stats_it - 1 ) - beta_k;
181  T abs_max_betas = beta_diff.template lpNorm< Eigen::Infinity >();
182 
183  T n_t = static_cast<T>( n );
184  T check_parameter = n_t*abs_max_betas / ( r_stats_it + rk );
185 
186  stats_cond &= ( check_parameter <= C );
187  }
188 
189  return stats_cond;
190 
191 }

225 T X_FOS< T >::dual_objective ( const MatrixT& X, \
226  const VectorT& Y, \
227  const VectorT& Beta, \
228  T r_stats_it ) {
229 
230  //Computation of s
231  T s_chunk = r_stats_it / ( 2.0*X.transpose()*( X*Beta - Y ) ).template lpNorm< Eigen::Infinity >();
232  T s_chunk_prime = ( - static_cast<T>( Y.transpose()*( X*Beta - Y ) ) )/( Y - X*Beta ).squaredNorm();
233  T s = std::min( std::max( - s_chunk, s_chunk_prime ), s_chunk );
234 
235  //Computation of nu tilde
236  VectorT nu_tilde = 2.0*s/r_stats_it*( X*Beta - Y );
237 
238  T d_nu = square( r_stats_it )/4.0*( nu_tilde + 2.0/r_stats_it*Y ).squaredNorm() - Y.squaredNorm();
239 
240  return d_nu;
241 }
