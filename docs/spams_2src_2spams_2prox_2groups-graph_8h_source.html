78 /* parse a line describing a node
79  id weight [ list-of-vars ] -> list-of-children
80  weight and list-of-children may be omitted; list-of-vars may be empty
81  Input :
82  s : C string to parse
83  vresult : result vector of 4 std::string (one per line component).
84  an empty element hans an empty string
85  Return:
86  -1 if error, 0 if OK
87  */
88 static inline int parse_line(char *s,std::vector<std::string> &vresult) {
89  char *p = skip_space(s);
90  if(p == NULL) return -1;
91  for(int i = 0;i < 4;i++) vresult.push_back(std::string(""));
92  // node id
93  char *p0 = p;
94  int n = 0;
95  while(! isspace(*p) && (*p != 0)) {
96  if(! isdigit(*p)) return -1;
97  p++;
98  n++;
99  }
100  vresult[0] = std::string(p0,n);
101  // weight
102  if((p = skip_space(p)) == NULL) return -1;
103  if(*p != '[') {
104  p0 = p;
105  n = 0;
106  while(! isspace(*p)) {
107  if(!isprint(*p)) return -1;
108  p++;
109  n++;
110  }
111  vresult[1] = std::string(p0,n);
112  }
113  // variables list
114  if((p = skip_space(p)) == NULL) return -1;
115  if(*p != '[') return -1;
116  p++;
117  if((p = skip_space(p)) == NULL) return -1;
118  p0 = p;
119  n = 0;
120  while(*p != ']') {
121  if(*p == 0) return -1;
122  p++;
123  n++;
124  }
125  if (n > 0) {
126  char *p1 = p - 1;
127  while(isspace(*p1)) {
128  p1--;
129  n--;
130  }
131  vresult[2] = std::string(p0,n);
132  }
133  p++;
134  // children list
135  if((p = skip_space(p)) == NULL) return 0;
136  if(*p != '-' || *(p+1) != '>') return -1;
137  p += 2;
138  if((p = skip_space(p)) == NULL) return 0;
139  // children list is not empty
140  p0 = p;
141  n = 0;
142  while (*p != 0) {
143  p++;
144  n++;
145  }
146  p--;
147  while(isspace(*p)) {
148  p--;
149  n--;
150  }
151  vresult[3] = std::string(p0,n);
152  return 0;
153 }

177 /*
178  Reads a text file describing "simply" the structure of groups
179  of variables needed by mexProximalGraph, mexProximalTree, mexFistaGraph,
180  mexFistaTree and mexStructTrainDL
181  Ecah line describes a group of variables as a node of a tree.
182  Each line has up to 4 fields separated by spaces:
183  node-id node-weight [variables-list] -> children-list
184  Let's define Ng = number of groups, and Nv = number of variables.
185  node-id must be in the range (0 - Ng-1), and there must be Ng nodes
186  weight is a float (1. if omitted)
187  variables-list : a space separated list of integers, maybe empty,
188  but '[' and '] must be present. Numbers in the range (0 - Nv-1)
189  children-list : a space separated list of node-id's
190  If the list is empty, '->' may be omitted.
191  The data must obey some rules :
192  A group contains the variables of the corresponding node and of the whole subtree.
193  Variables attached to a node are those that are not int the subtree.
194  If the data destination is a Graph, there may be several independant trees,
195  and a varibale may appear in several trees.
196  If the destination is a Tree, there must be only one tree, the root node
197  must have id == 0 and each variable must appear only once.
198  Args: filename
199  Return : a vector of StructNodeElem<T>
200 */

209  // while(fgets(buffer,1024,f) != NULL) {
210  if(buffer[0] == '#' || buffer[0] == 0) continue;
211  std::vector<string> lst;

266  nb *= degrees[i - 1];
267  level_starts[i] = k;
268  k += nb;
269  }

276  k = 0;
277  for(int id = 0;id < nb_levels;id++) {
278  if ((id + 1) < nb_levels)
279  k = level_starts[id + 1];
280  std::vector<int> lstx;
281  for(std::vector<int>::iterator it = lst.begin();it < lst.end();it++) {
282  int inode = *it;
283  std::vector<int> *children = new std::vector<int>;
284  for (int j = 0;j < degrees[id];j++) {
285  children->push_back(k);
286  lstx.push_back(k);
287  k++;
288  }
289  StructNodeElem<T> *node = new StructNodeElem<T>(inode,1.,new std::vector<int>(1,inode),children);
290  gstruct->push_back(node);
291  }
292  lst = lstx;
293  }
294  delete []level_starts;
295  return gstruct;
296 }
