143 T FOS<T>::DualityGap( uint r_stats_it ) {
144 
145  T rStatsIt_f = static_cast<T>( r_stats_it );
146 
147  //Duality Gap Set-Up
148  auto beta_t = Betas.col( statsIt - 1 );
149 
150  auto x_beta_t_prod = X * beta_t;
151  auto error = x_beta_t_prod - Y;
152 
153  auto x_cross_error = X.transpose() * error;
154  auto twice_x_cross_error = 2.0 * x_cross_error;
155 
156  auto alternative = rStatsIt_f/( twice_x_cross_error.template lpNorm< Eigen::Infinity >() );
157  T alt = static_cast<T>( alternative );
158 
159  T y_cross_error = static_cast<T>( Y.transpose() * error );
160  auto negative_error = Y - x_beta_t_prod;
161 
162  auto alternative_2 = ( -1.0 / negative_error.squaredNorm() ) * y_cross_error;
163 
164  T alt_2 = static_cast<T>( alternative_2 );
165 
166  auto s = std::min( std::max( alt, alt_2 ), -1.0*alt );
167  auto nu_t = -1.0 * ( 2 * s / rStatsIt_f ) * error;
168 
169  //Compute Duality Gap
170  auto f_beta = error.squaredNorm() + rStatsIt_f * beta_t.template lpNorm < 1 >() ;
171 
172  auto ret_val = nu_t + ( 2.0 / rStatsIt_f ) * Y;
173  auto d_nu = -0.25* square( rStatsIt_f ) * ret_val.squaredNorm() - Y.squaredNorm();
174 
175  auto duality_gap = static_cast<T>( f_beta ) - static_cast<T>( d_nu );
176  DEBUG_PRINT( "Duality Gap = " << duality_gap );
177 
178 
179  return duality_gap;
