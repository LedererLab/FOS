220 T FOS<T>::DualityGap( uint r_stats_it ) {
221 
222  T rStatsIt_f = static_cast<T>( r_stats_it );
223 
224  //Duality Gap Set-Up
225  auto beta_t = Betas.col( statsIt - 1 );
226 
227  auto x_beta_t_prod = X * beta_t;
228  auto error = x_beta_t_prod - Y;
229 
230  auto x_cross_error = X.transpose() * error;
231  auto twice_x_cross_error = 2.0 * x_cross_error;
232 
233  auto alternative = rStatsIt_f/( twice_x_cross_error.template lpNorm< Eigen::Infinity >() );
234  T alt = static_cast<T>( alternative );
235 
236  T y_cross_error = static_cast<T>( Y.transpose() * error );
237  auto negative_error = Y - x_beta_t_prod;
238 
239  auto alternative_2 = ( -1.0 / negative_error.squaredNorm() ) * y_cross_error;
240 
241  T alt_2 = static_cast<T>( alternative_2 );
242 
243  auto s = std::min( std::max( alt, alt_2 ), -1.0*alt );
244  auto nu_t = -1.0 * ( 2 * s / rStatsIt_f ) * error;
245 
246  //Compute Duality Gap
247  auto f_beta = error.squaredNorm() + rStatsIt_f * beta_t.template lpNorm < 1 >() ;
248 
249  auto ret_val = nu_t + ( 2.0 / rStatsIt_f ) * Y;
250  auto d_nu = -0.25* square( rStatsIt_f ) * ret_val.squaredNorm() - Y.squaredNorm();
251 
252  auto duality_gap = static_cast<T>( f_beta ) - static_cast<T>( d_nu );
253  DEBUG_PRINT( "Duality Gap = " << duality_gap );
254 
255 
256  return duality_gap;
257
